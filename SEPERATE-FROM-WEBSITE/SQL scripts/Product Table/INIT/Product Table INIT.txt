CREATE TABLE products (
    product_id INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
    product_name VARCHAR(20) NOT NULL,
    price DECIMAL (10, 2) NOT NULL,
    description VARCHAR(255) NOT NULL,
    thumbnail VARCHAR(255) NOT NULL,
    rating DECIMAL (2, 2),
    availability_status VARCHAR(20)
);

-- Product images table
CREATE TABLE product_images (
    image_id INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
    product_id INT NOT NULL,  -- Foreign key referencing the products table
    image_url VARCHAR(255) NOT NULL,  -- Store the image URL/path
    FOREIGN KEY (product_id) REFERENCES products(product_id) ON DELETE CASCADE
);

CREATE TABLE product_tags (
    tag_id INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
    product_id INT NOT NULL,
    tag VARCHAR(20) NOT NULL,
    FOREIGN KEY (product_id) REFERENCES products(product_id) ON DELETE CASCADE
);

CREATE TABLE product_reviews (
    review_id INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
    product_id INT NOT NULL,
    review VARCHAR(20) NOT NULL,
    review_date DATETIME NOT NULL,
    reviewer_email VARCHAR(125),
    reviewer_fullname VARCHAR(60),
    FOREIGN KEY (product_id) REFERENCES products(product_id) ON DELETE CASCADE
);
